1.OpenClaw 网关自动化部署校验脚本
1.1 脚本名称：openclaw_env.sh

#!/bin/bash
#OpenClaw网关一键部署、配置修改、启停、端口校验脚本
#目标服务器：10.0.8.10

#通过ssh远程登录网关主机，批量执行部署校验指令
ssh ubuntu@10.0.8.10 << EOF
#切换root权限
sudo -i
#修改网关监听模式为lan（内网全网卡监听）
openclaw config set gateway.bind lan
#停止正在运行的旧网关进程
openclaw gateway stop
#后台启动网关服务，&代表后台运行不阻塞后续命令
openclaw gateway run &
#休眠2秒，等待网关进程完全启动
sleep 2
#校验网关监听端口11948，有输出即代表服务正常运行
ss -tulpn | grep 11948
EOF
